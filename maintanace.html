Required Changes: Update your SplashActivity.kt with this code to check both version name and code

/** --------- UPDATE CHECK --------- */
private fun checkUpdate() {
    db.child("updateConfig").get().addOnSuccessListener { snap ->
        val minVersionName = snap.child("minimumVersionName").getValue(String::class.java) ?: "1.0.0"
        val minVersionCode = snap.child("minimumVersionCode").getValue(Int::class.java) ?: 1
        val updateMsg = snap.child("updateMessage").getValue(String::class.java) ?: "Update required."
        val updateLink = snap.child("updateLink").getValue(String::class.java) ?: "https://play.google.com/"

        val currentVersionName = try {
            packageManager.getPackageInfo(packageName, 0).versionName
        } catch (e: Exception) {
            "1.0.0"
        }
        
        val currentVersionCode = try {
            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P) {
                packageManager.getPackageInfo(packageName, 0).longVersionCode.toInt()
            } else {
                @Suppress("DEPRECATION")
                packageManager.getPackageInfo(packageName, 0).versionCode
            }
        } catch (e: Exception) {
            1
        }

        // Compare both version name and code
        val isVersionNameOutdated = compareVersions(currentVersionName, minVersionName) < 0
        val isVersionCodeOutdated = currentVersionCode < minVersionCode
        
        if (isVersionNameOutdated && isVersionCodeOutdated) {
            showUpdateDialog(updateMsg, updateLink)
        } else {
            checkMaintenance()
        }
    }.addOnFailureListener {
        checkMaintenance()
    }
}

/** --------- VERSION COMPARISON --------- */
private fun compareVersions(version1: String, version2: String): Int {
    val parts1 = version1.split(".")
    val parts2 = version2.split(".")
    
    val maxLength = maxOf(parts1.size, parts2.size)
    
    for (i in 0 until maxLength) {
        val part1 = if (i < parts1.size) parts1[i].toInt() else 0
        val part2 = if (i < parts2.size) parts2[i].toInt() else 0
        
        if (part1 < part2) return -1
        if (part1 > part2) return 1
    }
    
    return 0
}
